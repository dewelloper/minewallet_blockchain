function decryptMnemonic (seedHex, cipher) {
  if (cipher) {
    return BIP39.entropyToMnemonic(cipher(seedHex));
  } else {
    if (Helpers.isSeedHex(seedHex)) {
      return BIP39.entropyToMnemonic(seedHex);

}
